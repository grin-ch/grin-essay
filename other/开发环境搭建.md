# 配置网络

```shell
# 设置静态ip
vi /etc/sysconfig/network-scripts/ifcfg-ens33
```

**配置文件内容**

```sh
TYPE=Ethernet
PROXY_METHOD=none
BROWSER_ONLY=no
BOOTPROTO=static # static
DEFROUTE=yes
IPV4_FAILURE_FATAL=no
IPV6INIT=yes
IPV6_AUTOCONF=yes
IPV6_DEFROUTE=yes
IPV6_FAILURE_FATAL=no
IPV6_ADDR_GEN_MODE=stable-privacy
NAME=ens33
#UUID=82494bd6-0f41-4a42-ba92-4d364d68be48
DEVICE=ens33
ONBOOT=yes # yes
# 新增
GATEWAY=192.168.133.2
NETMASK=255.255.255.0
IPADDR=192.168.133.128
DNS1=223.5.5.5
DNS2=119.29.29.29
```

```shell
# 重启网络,然后通过ssh登入
service network restart
```

# 换源

```shell
# 安装wget
yum -y install wget

# 备份源
mv /etc/yum.repos.d/CentOS-Base.repo /etc/yum.repos.d/CentOS-Base.repo.bak
# 下载阿里源
wget -O /etc/yum.repos.d/CentOS-Base.repo http://mirrors.aliyun.com/repo/Centos-7.repo
# 清空/重建/更新缓存
yum clean all && yum makecache && yum upgrade
```

## docker

```shell
# 安装依赖软件
yum install -y yum-utils device-mapper-persistent-data lvm2
# 添加docker源(二选一就行)
# yum-config-manager --add-repo http://download.docker.com/linux/centos/docker-ce.repo
yum-config-manager --add-repo http://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo


# 查看可用版本, ps: 3:开头的都是精简版
yum list docker-ce --showduplicates | sort -r
# 指定版本安装
yum -y install docker-ce-18.03.1.ce-1.el7.centos
# 启动docker 并设置开机自启
systemctl start docker && systemctl enable docker

# docker配置镜像源
vim /etc/docker/daemon.json
# 写入 /etc/docker/daemon.json 文件内容
# 重载配置并且重启docker
systemctl daemon-reload && systemctl restart docker
```

**/etc/docker/daemon.json**

```json
{
    "registry-mirrors":[
        "https://hub-mirror.c.163.com",
        "https://registry.docker-cn.com",
        "https://docker.mirrors.ustc.edu.cn"
    ]
}
```

sudo curl -L "https://github.com/docker/compose/releases/download/v2.2.2/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose

## docker常用容器配置

```yaml
version: "3"

services:
    jaeger:
        image: "jaegertracing/all-in-one:latest"
        restart: always
        environment:
            COLLECTOR_ZIPKIN_HTTP_PORT: 9411
        ports:
            - "6831:6831/udp"
            - "6832:6832/udp"
            - "5775:5775/udp"
            - "5778:5778"
            - "16686:16686"
            - "14268:14268"
            - "9411:9411"

    redis:
        image: "redis:latest"
        restart: always
        ports:
            - "6379:6379"
        volumes:
            - "$HOME/data/redis/data:/data"
            - "$HOME/data/redis/redis.conf:/usr/local/etc/redis/redis.conf"
        command:
            redis-server /usr/local/etc/redis/redis.conf

    pgsql:
        image: "postgres:latest"
        privileged: true
        restart: always
        environment:
            POSTGRES_PASSWORD: "rootroot"
        ports:
            - "5432:5432"
        volumes:
            - "$HOME/data/pgsql/data:/var/lib/postgresql/data"

    etcd:
        image: "bitnami/etcd:latest"
        restart: always
        environment:
            ETCD_ADVERTISE_CLIENT_URLS: "http://etcd-server:2379"
            ALLOW_NONE_AUTHENTICATION: "yes"
        ports:
            - "2379:2379"
            - "2380:2380"
        volumes:
            - "$HOME/data/etcd/data:/etcd-data"

```
